    var str = "A long " + "string to con" + "catenate to" + "gether, mate.";

    document.write("String: " + str + "<br />");

    // write the strings length
    document.write("String Length: " + str.length + "<br />");

    // write the strings starting index for the word mate
    document.write("String for 'mate': " + str.indexOf("mate") + "<br />");

    // if you only need a small slice of your string
    document.write("String slice for string: " + str.slice(7, 14) + "<br />");

    // if you want to slice your string from one starting point to the end
    document.write("String slice for s: " + str.slice(7) + "<br />");

    // if you only need a small slice of your string
    document.write("String substring using index, length: " + str.substr(7, 7) + "<br />");

    // replace strings with other strings
    document.write("String replacement: " + str.replace("concatenate", "join") + "<br />");

    // find char at an index
    document.write("String index(2): " + str.charAt(2) + "<br />");

    // replace all
    var str2 = str.split(" ");

    document.write("str2 is now an array, which is why it has commas after the words: " + str2 + "<br />");

    // if you need to trim some white space
    var str3 = str.trim();
